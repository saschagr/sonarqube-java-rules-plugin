This rule detects non documented private methods

Noncompliant Code Example

private String doIt() {
   //do it
   return "doIt";
}

private void doItNow(String foo) {
   //do it now
}

Compliant Solution

/**
 * Do it.
 * @return the result value.
 */
private String doIt() {
   //do it
   return "doIt";
}

/**
 * Do it now and for ever.
 * @param foo the foo value to the job now.
 */
private void doItNow(String foo) {
   //do it now
}
